# 自动获取日对账单操作流程及接口文档

更新时间：2025-10-26 14:56:44

### 

一、 操作流程

1. 入驻开放平台

2\. 店铺后台对账中心—货款账单页面可见日账单权限申请入口

（注：仅在店铺授权具有日账单接口权限的商家自研应用后可见此申请入口）

![](https://t16img.yangkeduo.com/eden-admin/2021-09-06/cb2c4bae-aa05-4688-8d85-e25b300c7872.png)

3. 接入自动获取货款对账单接口（接口文档见下文）

4. 调用接口可获取日对账单下载链接，下载文件格式为压缩加密文件

5. 通过MMS后台获取解密密钥

（1）在商家后台左侧导航栏【账户资金】-【对账中心】-【货款账户】页面右侧有【获取日账单密钥】文字链

![](https://t16img.yangkeduo.com/perse-open-api/2023-12-27/3f584804-5633-4138-bb9e-19adc632201e.png)

（2） 点击文字链，跳出身份验证弹窗，通过店铺管理人手机号获取验证码验证身份

（3） 验证身份成功后，弹窗出现密钥，可一键复制密钥用于对账单文件的解密

注：

（1） 新密钥生成后次日起生效，且只对次日及以后的对账单文件有效

（2） 密钥的生效期是固定的，历史文件只能使用当时对应生效的密钥解密；商家可手动更新密钥，更新密钥只对之后生成的文件有效，暂不做自动更新逻辑

（3） 每日10点后允许获取前一日对账单

6. 用密钥先对加密压缩文件进行解密操作，然后对解密之后的文件进行解压缩

解密程序demo：

 package decrypt;

 import javax.crypto.Cipher;

 import javax.crypto.CipherOutputStream;

 import javax.crypto.KeyGenerator;

 import javax.crypto.SecretKey;

 import javax.crypto.spec.SecretKeySpec;

 import java.io.File;

 import java.io.FileInputStream;

 import java.io.FileOutputStream;

 import java.security.SecureRandom;

 public class DecryptFileUtil {

  private static final String ALGORITHM \= "AES";

  private static final int KEY\_SIZE \= 128;

  private static final int CACHE\_SIZE \= 1024;

  /\*\*

  \* <p>

  \* 文件解密

  \* </p\>

  \*

  \* @param seed 密钥，从商家后台获取

  \* @param sourceFilePath 源文件路径，源文件指从下载链接下载下来的文件

  \* @param destFilePath 目标文件路径，目标文件指对源文件进行解密操作之后的文件

  \* @throws Exception

  \*/

  public static void decryptFile(String seed, String sourceFilePath, String destFilePath) throws Exception {

   if (seed \== null || seed.equals("")) {

   throw new Exception("密钥为空，解密失败");

   }

   KeyGenerator keyGenerator \= KeyGenerator.getInstance(ALGORITHM);

   SecureRandom secureRandom \= SecureRandom.getInstance("SHA1PRNG");

   secureRandom.setSeed(seed.getBytes());

   keyGenerator.init(KEY\_SIZE, secureRandom);

   SecretKey secretKey \= keyGenerator.generateKey();

   byte\[\] raw \= secretKey.getEncoded();

   SecretKeySpec secretKeySpec \= new SecretKeySpec(raw, ALGORITHM);

   File sourceFile \= new File(sourceFilePath);

   File destFile \= new File(destFilePath);

   if (sourceFile.exists() && sourceFile.isFile()) {

    if (!destFile.getParentFile().exists()) {

   destFile.getParentFile().mkdirs();

   }

   destFile.createNewFile();

   FileInputStream in \= new FileInputStream(sourceFile);

   FileOutputStream out \= new FileOutputStream(destFile);

   Cipher cipher \= Cipher.getInstance("AES/ECB/PKCS5Padding");

   cipher.init(Cipher.DECRYPT\_MODE, secretKeySpec);

   CipherOutputStream cout \= new CipherOutputStream(out, cipher);

   byte\[\] cache \= new byte\[CACHE\_SIZE\];

   int nRead \= 0;

   while ((nRead \= in.read(cache)) != \-1) {

   cout.write(cache, 0, nRead);

   cout.flush();

   }

   cout.close();

   out.close();

   in.close();

   }

  }

 }
